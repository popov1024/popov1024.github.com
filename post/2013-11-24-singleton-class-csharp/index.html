По методу Джеффри Рихтера использую такой singleton:

using System; using System.Threading; public sealed class singletonclass { private static readonly Object SLock = new Object(); private static singletonclass _instance; private readonly int bar; private singletonclass() { bar = 123; } public static singletonclass Instance { get { if (_instance != null) return _instance; Monitor.Enter(SLock); var temp = new singletonclass(); Interlocked.Exchange(ref _instance, temp); Monitor.Exit(SLock); return _instance; } } public int foo() { return bar; } }
