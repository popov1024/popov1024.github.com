1С Web-Cервис -- Определение типа

Вот такая структура типов обычно ставит в тупик из-за недостаточно четкого описание объектной модели типизации XDTO.

Обычно такая ситуация возникает в таких ситуациях: когда описание делается вложено, а не разбито на отдельные сущности.

Пример программы формирования объекта данной структуры:

ws = WSСсылки.WSСсылка.СоздатьWSПрокси("http://xmlns.oracle.com/orawsv/OWNER_OIL/COMPLEXWS", "COMPLEXWSService", "COMPLEXWSPort");
ОпределениеТипа = ws.ФабрикаXDTO.Тип("http://xmlns.oracle.com/orawsv/OWNER_OIL/COMPLEXWS", "CMPLX_DATAType");
Вх = ws.ФабрикаXDTO.Создать(ОпределениеТипа);
ТипCMPLX_DATA = ОпределениеТипа.Свойства.Получить("CMPLX_DATA").Тип;
Вх.CMPLX_DATA = ws.ФабрикаXDTO.Создать(ТипCMPLX_DATA);
Вх.CMPLX_DATA.NN = 123;
Вх.CMPLX_DATA.DD = ТекущаяДата();
Вх.CMPLX_DATA.TT = ТекущаяДата();
Вх.CMPLX_DATA.SS = ws.ФабрикаXDTO.Создать(ТипCMPLX_DATA.Свойства.Получить("SS").Тип, "пример строки");
Вх.CMPLX_DATA.RECS = ws.ФабрикаXDTO.Создать(ТипCMPLX_DATA.Свойства.Получить("RECS").Тип);
ТипСтрокаОбъекта = ws.ФабрикаXDTO.Тип("http://xmlns.oracle.com/orawsv/OWNER_OIL/COMPLEXWS", "CMPLX_REC_IntType");

СтрокаОбъекта = ws.ФабрикаXDTO.Создать(ТипСтрокаОбъекта);
СтрокаОбъекта.N = 123;
СтрокаОбъекта.S = ws.ФабрикаXDTO.Создать(ТипСтрокаОбъекта.Свойства.Получить("S").Тип, "первая строка");
СтрокаОбъекта.D = ТекущаяДата();
СтрокаОбъекта.T = ТекущаяДата();
    
СтрокаОбъекта2 = ws.ФабрикаXDTO.Создать(ТипСтрокаОбъекта);
СтрокаОбъекта2.N = 345;
СтрокаОбъекта2.S = ws.ФабрикаXDTO.Создать(ТипСтрокаОбъекта.Свойства.Получить("S").Тип, "вторая строка");
СтрокаОбъекта2.D = ТекущаяДата();
СтрокаОбъекта2.T = ТекущаяДата();
    
Вх.CMPLX_DATA.RECS.CMPLX_REC.Добавить(СтрокаОбъекта);
Вх.CMPLX_DATA.RECS.CMPLX_REC.Добавить(СтрокаОбъекта2);

res = ws.COMPLEXWS(Вх);
